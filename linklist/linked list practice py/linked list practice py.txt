# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def deleteNode(self, node):
        """
        :type node: ListNode
        :rtype: void Do not return anything, modify node in-place instead.
        """
        """
        Input: head = [4,5,1,9], node = 5
        Output: [4,1,9]
        
        search for 5
        in first node, is value 5? 
        no!
        go to next node
        in next node, is value 5? 
        yes!
        
        delete this node
        copy value of NEXT node, into value of THIS node
        make this node point to NEXT NEXT node
        
        """
        if self.val == node.val:
            del self
        else:    
            deleteNode(node.next)
